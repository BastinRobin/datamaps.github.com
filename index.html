DataMaps

Maps built for data visualizations.

Scalable visualizations with SVG

Requires a single Javascript file

Add data to create choropleth maps

$('#map_click').datamaps({
      fills: {
        'REP': '#CC4731',
        'DEM': '#306596',
        'HEAVY_DEM': '#667FAF',
        /*... snipped */
        defaultFill: '#EDDC4E'
      },
      data:{
        "AZ": {
            "fillKey": "REP",
            "electoralVotes": 5
        },
        "CO": {
            "fillKey": "LIGHT_DEM",
            "electoralVotes": 5
        } /* ... snipped */
})
